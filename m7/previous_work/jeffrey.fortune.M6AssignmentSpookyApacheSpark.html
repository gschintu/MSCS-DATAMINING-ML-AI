# Balance the data so authors have the same number of rows
author_counts = train_data.groupBy("author").agg(count("*").alias("count"))
min_count = author_counts.agg({"count": "min"}).collect()[0][0]
authors = [row["author"] for row in author_counts.collect()]
balanced_data = spark.createDataFrame([], train_data.schema)
for author in authors:
    author_data = train_data.filter(col("author") == author).orderBy("id").limit(min_count)
    balanced_data = balanced_data.union(author_data)
print("\nBalanced rows for the author column:")
balanced_data.groupBy("author").count().show()

# Remove punctuation from the text column
balanced_data = balanced_data.withColumn("clean_text", regexp_replace(col("text"), "[^a-zA-Z0-9\s]", ""))

# Using the lemmatizer from nltk, we create tokens and lemmatize them.
lemmatizer = WordNetLemmatizer()
lemmatize_udf = udf(
    lambda text: [lemmatizer.lemmatize(token.lower()) for token in text.split()], 
    ArrayType(StringType())
)
balanced_data = balanced_data.withColumn("tokens", lemmatize_udf("clean_text"))

# Remove the stop words
stop_words = StopWordsRemover.loadDefaultStopWords("english")
remover = StopWordsRemover(inputCol="tokens", outputCol="filtered_tokens", stopWords=stop_words)
balanced_data = remover.transform(balanced_data)

balanced_data.select("author", "filtered_tokens").show(truncate=False)

Stage 2 Feature Extraction¶

  • Use CountVectorizer to create a bag of words
  • IDF to create a TF-IDF matrix
  • Normalizer to normalize the TF-IDF matrix
  • Pipeline to build and execute stages.
In [39]:
# Step 3: TF-IDF calculation
cv = CountVectorizer(inputCol="filtered_tokens", outputCol="vectorized_tokens")
idf = IDF(inputCol="vectorized_tokens", outputCol="tfidf")
# Step 4: Normalization
normalizer = Normalizer(inputCol="tfidf", outputCol="normalized_features")
author_labeler = StringIndexer(inputCol="author", outputCol="author_label").setHandleInvalid("keep")
# Step 5: Create pipeline for chaining the text mining transformer
pipeline = Pipeline(stages=[cv, idf, normalizer, author_labeler])
# Step 6: Apply the pipeline to frame
processed_data = pipeline.fit(balanced_data).transform(balanced_data)
# Convert author_label from float to int
processed_data = processed_data.withColumn("author_label", col("author_label").cast("integer"))
# Print the normalized features with labels
processed_data.select("author", "author_label", "normalized_features").show(truncate=True)

# Add an ID column to the data
windowSpec = Window.partitionBy("author").orderBy("id")
processed_data_with_id = processed_data.withColumn("id", row_number().over(windowSpec))

# Group the data by author and get the count of each author
author_counts = processed_data_with_id.groupBy("author").count()
total_test_proportion = 0.2
test_sizes = author_counts.select("author", (col("count") * total_test_proportion).alias("holdout"))
data_with_test_sizes = processed_data_with_id.join(test_sizes, "author")

# Split the data into train and test based on the calculated test sizes
train_data = data_with_test_sizes.filter(col("id") > col("holdout")).drop("holdout", "id")
test_data = data_with_test_sizes.filter(col("id") <= col("holdout")).drop("holdout", "id")
train_data = train_data.withColumn("author_label", col("author_label").cast("double"))
